# -*- coding: utf-8 -*-

import numpy as np

def to_image(L):             #将1*2400连续胎心率数据点转化成120*2400胎心率曲线图
    image=np.zeros(120,2400)
    for i in range(1,2401):  #不要第一位标志位和最后一位冗余位
        if(L[i]<80):
            L[i]=80
        elif(L[i]>200):
            L[i]=199
        image[L[i]-80][i-1]=1
    return image
