def data_seperate(filename,number): #seperate data into training and test. argu:number means the number of data.
    fileobject = open(filename+'.txt', "r")
    trainingdata_fileobject=open('trainingdata.txt',"w")
    testdata_fileobject=open('testingdata.txt',"w")
    
    for i in range(number):
        fetal_data=fileobject.readline()
        if(i<int(0.7*number)):
            trainingdata_fileobject.write("%s"%fetal_data)
        else:
            testdata_fileobject.write("%s"%fetal_data)
    
    return True
